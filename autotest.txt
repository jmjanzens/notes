### to setup
$ mkdir b && cd b // make build dir
$ ../configure --[flag(s)]
$ make -j 10
// then test with '$ make check'

### run test suite
make check TESTSUITEFLAGS="-k <testgroup> --verbose"

### if testgroup out of synch with file being tested
$ touch testsuite.at

### to regen testsuite file
$ make check // in scratch dir

### to run just one specific check (with more detail)
//or
make check TESTSUITEFLAGS="-k <testgroup> --verbose"

### run distcheck
make distcheck
//or
make distcheck DISTCHECK_CONFIGURE_FLAGS="--<flag(s)"

### run tests with spaces in keywords
// place single quotes around keyword
// i.e., for AT_KEYWORDS([packet firmwareupdate])
// run this test like so:
$  make check TESTSUITEFLAGS="-k 'packet firmwareupdate' --verbose"
